CREATE DATABASE ejercicio3;
USE ejercicio3;
CREATE TABLE libros( ISBN VARCHAR(20) PRIMARY KEY, titulo VARCHAR(40), fechaEdicion DATE, numPrestamos INT );
CREATE TABLE autores( idAutor VARCHAR(20) PRIMARY KEY, nomAutor VARCHAR(20), estilo VARCHAR(20), nacionalidad VARCHAR(30), fechaNac DATE, numPerstamos INT );
CREATE TABLE escriben ( idAutor VARCHAR(20) PRIMARY KEY, ISBN VARCHAR(20) );
CREATE TABLE socios( idSocio VARCHAR(20) PRIMARY KEY, nomSoc VARCHAR(20), dirSoc VARCHAR(20), tefno INT, fechaFinSancion DATE, numLibrosLeidos INT );
CREATE TABLE ejemplares( codigoEjemplar VARCHAR(20) PRIMARY KEY, estadoCons VARCHAR(20), ISBN VARCHAR (20) );
-- CHECK estadoCons (Excelente, Bueno, Malo)
CREATE TABLE prestan( idSocio VARCHAR(20) PRIMARY KEY, codigoEjemplar VARCHAR(20), fechaPrestamo DATE, fechaDevolucion DATE );
